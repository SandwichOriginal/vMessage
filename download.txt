// ─── ECDH Crypto ───────────────────────────────────────────────────
async function initECDH() {
  try {
    const privJwk = JSON.parse(localStorage.getItem(K.ECDH_PRIV) || "null");
    const pubJwk  = JSON.parse(localStorage.getItem(K.ECDH_PUB)  || "null");
    if (privJwk && pubJwk) {
      const priv = await crypto.subtle.importKey("jwk", privJwk, {name:"ECDH",namedCurve:"P-256"}, true, ["deriveKey","deriveBits"]);
      const pub  = await crypto.subtle.importKey("jwk", pubJwk,  {name:"ECDH",namedCurve:"P-256"}, true, []);
      myECDHPair   = { privateKey: priv, publicKey: pub };
      myECDHPubJwk = pubJwk;
      return;
    }
  } catch(e) { /* regenerate */ }

  myECDHPair = await crypto.subtle.generateKey({name:"ECDH",namedCurve:"P-256"}, true, ["deriveKey","deriveBits"]);
  myECDHPubJwk = await crypto.subtle.exportKey("jwk", myECDHPair.publicKey);
  const privJwk = await crypto.subtle.exportKey("jwk", myECDHPair.privateKey);
  localStorage.setItem(K.ECDH_PRIV, JSON.stringify(privJwk));
  localStorage.setItem(K.ECDH_PUB,  JSON.stringify(myECDHPubJwk));
}

async function deriveSharedKey(theirPubJwk) {
  const theirPub = await crypto.subtle.importKey("jwk", theirPubJwk, {name:"ECDH",namedCurve:"P-256"}, true, []);
  const bits = await crypto.subtle.deriveBits({name:"ECDH", public: theirPub}, myECDHPair.privateKey, 256);
  const hkdf = await crypto.subtle.importKey("raw", bits, "HKDF", false, ["deriveKey"]);
  const aes  = await crypto.subtle.deriveKey(
    { name:"HKDF", salt: new Uint8Array(32), info: enc.encode("vmsg-direct-v1"), hash:"SHA-256" },
    hkdf, {name:"AES-GCM", length:256}, true, ["encrypt","decrypt"]
  );
  const raw = await crypto.subtle.exportKey("raw", aes);
  return bufToB64(raw);
}

async function importAES(b64) {
  return crypto.subtle.importKey("raw", b64ToBuf(b64), {name:"AES-GCM"}, false, ["encrypt","decrypt"]);
}

async function encryptPayload(keyB64, obj) {
  const key = await importAES(keyB64);
  const iv  = crypto.getRandomValues(new Uint8Array(12));
  const ct  = await crypto.subtle.encrypt({name:"AES-GCM", iv}, key, enc.encode(JSON.stringify(obj)));
  return { iv: bufToB64(iv.buffer), ct: bufToB64(ct) };
}

async function decryptPayload(keyB64, envelope) {
  try {
    const key = await importAES(keyB64);
    const iv  = new Uint8Array(b64ToBuf(envelope.iv));
    const pt  = await crypto.subtle.decrypt({name:"AES-GCM", iv}, key, b64ToBuf(envelope.ct));
    return JSON.parse(dec.decode(pt));
  } catch { return null; }
}

    const meta   = document.createElement("div"); meta.className = "room-btn-meta";

// ─── Peer / Network ────────────────────────────────────────────────
function initPeerId() {
  let stored = localStorage.getItem(K.PEER_ID);
  if (!stored) { stored = "vmsg-d-" + uuid(); localStorage.setItem(K.PEER_ID, stored); }
  myPeerId = stored;
}

function isOnline(convId) { const c = openConns.get(convId); return !!(c && c.open); }

// Deterministic role: the peer whose ID sorts lower is ALWAYS the initiator.
// Both sides independently compute the same answer — zero timing hacks needed.
function amInitiator(conv) { return myPeerId < conv.peerDirectId; }

function startStablePeer() {
  if (stablePeer && !stablePeer.destroyed) return;
  stablePeer = new Peer(myPeerId);

  stablePeer.on("open", () => {
    updateStatus("Connected", "green");
    tryConnectAll();
  });

  stablePeer.on("connection", conn => {
    conn.on("open", () => handleIncomingConn(conn));
  });

  stablePeer.on("error", err => {
    if (err.type === "unavailable-id") {
      myPeerId = "vmsg-d-" + uuid();
      localStorage.setItem(K.PEER_ID, myPeerId);
      setTimeout(startStablePeer, 500);
    } else {
      updateStatus("Attempting to connect.", "yellow");
      setTimeout(startStablePeer, 4000);
    }
  });

  stablePeer.on("disconnected", () => {
    updateStatus("Reconnecting…", "yellow");
    try { stablePeer.reconnect(); } catch { startStablePeer(); }
  });
}

function handleIncomingConn(conn) {
  const conv = conversations.find(c => c.peerDirectId === conn.peer);
  if (!conv) { conn.close(); return; }

  // Accept incoming conn regardless of our role — handles edge cases where
  // both restarted simultaneously or the initiator role flipped.
  const old = openConns.get(conv.id);
  if (old && old !== conn) { try { old.close(); } catch {} }

  openConns.set(conv.id, conn);
  renderConversations();
  if (activeConvId === conv.id) setConvStatus(conv.peerName, "Online · end-to-end encrypted", true);

  conn.on("data",  async data => handleIncomingMsg(conv.id, data));
  conn.on("close", ()         => onPeerDisconnect(conv.id, conv.peerName));
  conn.on("error", ()         => onPeerDisconnect(conv.id, conv.peerName));
}

function onPeerDisconnect(convId, peerName) {
  openConns.delete(convId);
  renderConversations();
  if (activeConvId === convId) setConvStatus(peerName, "Offline · messages won't be delivered", true);
  // Only the initiator actively retries — the receiver just waits for an
  // incoming connection from the initiator.
  const conv = getConv(convId);
  if (conv && amInitiator(conv)) setTimeout(() => tryConnect(convId), 3000);
}

function tryConnectAll() {
  conversations.forEach(conv => { if (!isOnline(conv.id)) tryConnect(conv.id); });
}

function tryConnect(convId) {
  const conv = getConv(convId);
  if (!conv || !stablePeer || stablePeer.destroyed || stablePeer.disconnected) return;
  if (isOnline(convId)) return;

  // ONLY the designated initiator ever calls stablePeer.connect().
  // The receiver only accepts via handleIncomingConn.
  // This eliminates the simultaneous-connect race in PeerJS completely.
  if (!amInitiator(conv)) return;

  try {
    const conn = stablePeer.connect(conv.peerDirectId, {reliable: true});
    let resolved = false;

    const timeout = setTimeout(() => {
      if (resolved) return; resolved = true;
      try { conn.close(); } catch {}
    }, 8000);

    conn.on("open", () => {
      if (resolved) return; resolved = true; clearTimeout(timeout);
      openConns.set(convId, conn);
      renderConversations();
      if (activeConvId === convId) setConvStatus(conv.peerName, "Online · end-to-end encrypted", true);
      conn.on("data",  async data => handleIncomingMsg(convId, data));
      conn.on("close", ()         => onPeerDisconnect(convId, conv.peerName));
      conn.on("error", ()         => onPeerDisconnect(convId, conv.peerName));
    });
    conn.on("error", () => { if (resolved) return; resolved = true; clearTimeout(timeout); });
  } catch {}
}

async function handleIncomingMsg(convId, data) {
  if (!data || data.type !== "direct_msg") return;
  const conv = getConv(convId); if (!conv) return;

  const plain = await decryptPayload(conv.sharedKeyB64, data.enc);
  if (!plain || !plain.text || !plain.id) return;

  if (conv.messages.some(m => m.id === plain.id)) return;
  if (plain.time && plain.time > Date.now() + 60000) return;

  const msg = { id: plain.id, text: sanitize(plain.text, 1000), fromMe: false, time: plain.time || Date.now() };
  conv.messages.push(msg);
  saveConvos();
  renderConversations();

  if (activeConvId === convId) addChatLine({name: conv.peerName}, msg.text, false);
}

// ─── Pairing Modal ─────────────────────────────────────────────────
function generateCode() {
  const chars = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789";
  const arr = crypto.getRandomValues(new Uint8Array(8));
  return Array.from(arr).map(b => chars[b % chars.length]).join("");
}

let activePairCode = null;

function openPairModal() {
  if (pairPeer) { try { pairPeer.destroy(); } catch {} pairPeer = null; }
  if (pairTimer) { clearInterval(pairTimer); pairTimer = null; }

  activePairCode = generateCode();
  const display  = activePairCode.slice(0,4) + "-" + activePairCode.slice(4);

  $("pair-code-display").textContent  = display;
  $("pair-code-status").textContent   = "Waiting for connection…";
  $("pair-code-status").className     = "pair-code-status";
  $("pair-enter-input").value         = "";
  $("pair-enter-error").textContent   = "";
  $("pair-enter-btn").disabled        = false;
  $("pair-enter-btn").textContent     = "Connect";

  let remaining = 5 * 60;
  renderPairTimer(remaining);
  pairTimer = setInterval(() => {
    remaining--;
    renderPairTimer(remaining);
    if (remaining <= 0) {
      clearInterval(pairTimer); pairTimer = null;
      if (pairPeer) { try { pairPeer.destroy(); } catch {} pairPeer = null; }
      $("pair-code-status").textContent = "Code expired. Close and reopen to generate a new one.";
      $("pair-code-status").className   = "pair-code-status error";
    }
  }, 1000);

  pairPeer = new Peer("vmsg-pair-" + activePairCode);
  pairPeer.on("error", err => {
    if (err.type === "unavailable-id") {
      $("pair-code-status").textContent = "Code already in use, regenerating…";
      setTimeout(openPairModal, 400);
    } else {
      $("pair-code-status").textContent = "Network error. Check your connection.";
      $("pair-code-status").className   = "pair-code-status error";
    }
  });
  pairPeer.on("connection", conn => {
    conn.on("open", () => {
      $("pair-code-status").textContent = "Connected! Establishing secure channel…";
      $("pair-code-status").className   = "pair-code-status success";
      // Host side — isJoiner = false
      doPairHandshake(conn, false);
    });
  });

  $("pair-overlay").classList.remove("hidden");
}

function renderPairTimer(secs) {
  const m = Math.floor(secs / 60).toString().padStart(2,"0");
  const s = (secs % 60).toString().padStart(2,"0");
  $("pair-timer").textContent = `${m}:${s}`;
}

  if (!activePairCode) return;
  const display = activePairCode.slice(0,4) + "-" + activePairCode.slice(4);
  navigator.clipboard.writeText(display).then(() => {
    const btn = $("pair-copy-btn");
    btn.innerHTML = `<i data-lucide="check"></i>Copied!`;
    btn.classList.add("copied");
    if (window.lucide) lucide.createIcons({nodes:[btn]});
    setTimeout(() => {
      btn.innerHTML = `<i data-lucide="copy"></i>Copy code`;
      btn.classList.remove("copied");
      if (window.lucide) lucide.createIcons({nodes:[btn]});
    }, 2000);
  });
}

function closePairModal() {
  if (pairPeer)  { try { pairPeer.destroy();  } catch {} pairPeer  = null; }
  if (pairTimer) { clearInterval(pairTimer); pairTimer = null; }
  activePairCode = null;
  $("pair-overlay").classList.add("hidden");
}

async function enterPairCode() {
  let raw = $("pair-enter-input").value.trim().toUpperCase().replace(/-/g, "");
  $("pair-enter-error").textContent = "";

  if (raw.length !== 8) { $("pair-enter-error").textContent = "Code must be 8 characters."; return; }
  if (!stablePeer || stablePeer.destroyed || stablePeer.disconnected) {
    $("pair-enter-error").textContent = "Not connected to network. Wait a moment and try again."; return;
  }

  const btn = $("pair-enter-btn");
  btn.disabled = true; btn.textContent = "Connecting…";

  try {
    const conn = stablePeer.connect("vmsg-pair-" + raw, {reliable: true});
    let resolved = false;

    const timeout = setTimeout(() => {
      if (resolved) return; resolved = true;
      btn.disabled = false; btn.textContent = "Connect";
      $("pair-enter-error").textContent = "Couldn't reach that code. Is the other person waiting?";
      try { conn.close(); } catch {}
    }, 9000);

    conn.on("open", () => {
      if (resolved) return; resolved = true; clearTimeout(timeout);
      // Joiner side — isJoiner = true
      doPairHandshake(conn, true);
    });
    conn.on("error", () => {
      if (resolved) return; resolved = true; clearTimeout(timeout);
      btn.disabled = false; btn.textContent = "Connect";
      $("pair-enter-error").textContent = "Connection failed. Check the code and try again.";
    });
  } catch(e) {
    btn.disabled = false; btn.textContent = "Connect";
    $("pair-enter-error").textContent = "Failed to initiate connection.";
  }
}

// ─── Post-pair retry ───────────────────────────────────────────────
// Initiator retries every 2s for 30s until the real connection lands.
// Receiver's calls are no-ops (tryConnect gates on amInitiator).
function startPostPairRetry(convId) {
  const deadline = Date.now() + 30000;
  const attempt = () => {
    if (isOnline(convId) || Date.now() > deadline) return;
    tryConnect(convId);
    setTimeout(attempt, 2000);
  };
  setTimeout(attempt, 1000);
}

async function doPairHandshake(conn, isJoiner = false) {
  const hello = {
    type:      "pair_hello",
    directId:  myPeerId,
    name:      uname(),
    ecdhPub:   myECDHPubJwk,
    timestamp: Date.now(),
  };

  let theirHello = null;
  let mySent     = false;

  const finalize = async () => {
    if (!theirHello || !mySent) return;

    try {
      if (!theirHello.directId || !theirHello.ecdhPub || typeof theirHello.name !== "string") {
        throw new Error("Invalid pair_hello");
      }
      if (theirHello.directId === myPeerId) throw new Error("Cannot pair with yourself");

      const sharedKeyB64 = await deriveSharedKey(theirHello.ecdhPub);

      const existing = conversations.find(c => c.peerDirectId === theirHello.directId);
      const convId   = existing ? existing.id : uuid();

      if (existing) {
        existing.peerName       = sanitize(theirHello.name, 30) || "Anonymous";
        existing.sharedKeyB64   = sharedKeyB64;
        existing.peerECDHPubJwk = theirHello.ecdhPub;
        saveConvos();
        closePairModal();
        selectConversation(existing.id);
        addChatLine("System", `Re-paired with ${existing.peerName}. Encryption key refreshed.`, true);
      } else {
        const newConv = {
          id:             convId,
          peerDirectId:   theirHello.directId,
          peerName:       sanitize(theirHello.name, 30) || "Anonymous",
          peerECDHPubJwk: theirHello.ecdhPub,
          sharedKeyB64,
          createdAt:      Date.now(),
          messages:       [],
        };
        conversations.unshift(newConv);
        saveConvos();
        closePairModal();
        selectConversation(convId);
        addChatLine("System", `You paired with ${newConv.peerName}. All messages are end-to-end encrypted.`, true);
      }

      $("pair-enter-btn").disabled = false;
      $("pair-enter-btn").textContent = "Connect";

      // Close the pairing connection, then let deterministic roles take over.
      // The initiator (lower peerId) will connect; receiver just waits.
      try { conn.close(); } catch {}
      startPostPairRetry(convId);

    } catch(err) {
      $("pair-code-status").textContent = "Pairing failed: " + (err.message || "Unknown error");
      $("pair-code-status").className   = "pair-code-status error";
      $("pair-enter-error").textContent = "Pairing failed. Please try again.";
      $("pair-enter-btn").disabled = false;
      $("pair-enter-btn").textContent = "Connect";
      try { conn.close(); } catch {}
    }
  };

  conn.on("data", async data => {
    if (data && data.type === "pair_hello") {
      theirHello = data;
      await finalize();
    }
  });

  conn.send(hello);
  mySent = true;
  await finalize();
}

// ─── Conversation Actions ──────────────────────────────────────────
function selectConversation(convId) {
  const conv = getConv(convId); if (!conv) return;
  activeConvId = convId;

  clearChat();
  const w = $("chat-welcome"); if (w) w.remove();

  const online = isOnline(convId);
  setConvStatus(conv.peerName, online ? "Online · end-to-end encrypted" : "Offline · messages won't be delivered", true);

  conv.messages.forEach(msg => {
    if (msg.fromMe) addChatLine("Me", msg.text, false);
    else addChatLine({name: conv.peerName}, msg.text, false);
  });

  renderConversations();
  if (!online) tryConnect(convId);
}

function leaveConversation() {
  activeConvId = null;
  clearChat();
  setConvStatus("Not connected", "Select a conversation to begin", false);
  renderConversations();
}

function deleteConversation() {
  if (!activeConvId) return;
  const conv = getConv(activeConvId);
  if (!conv) return;

  const confirmed = confirm(`Delete conversation with ${conv.peerName}? This removes all messages and cannot be undone.`);
  if (!confirmed) return;

  const conn = openConns.get(activeConvId);
  if (conn) { try { conn.close(); } catch {} openConns.delete(activeConvId); }

  conversations = conversations.filter(c => c.id !== activeConvId);
  saveConvos();
  activeConvId = null;
  clearChat();
  setConvStatus("Not connected", "Select a conversation to begin", false);
  renderConversations();
}

// ─── Sending Messages ──────────────────────────────────────────────
async function sendDirectMsg() {
  const input = $("msg");
  const text  = sanitize(input.value.trim(), 1000);
  if (!text || !activeConvId) return;

  const conv = getConv(activeConvId); if (!conv) return;
  const conn = openConns.get(activeConvId);

  const msgId = uuid();
  const msg   = { id: msgId, text, fromMe: true, time: Date.now(), status: "sent" };

  conv.messages.push(msg);
  saveConvos();
  addChatLine("Me", text, false);
  input.value = "";
  $("sendBtn").disabled = true;
  renderConversations();

  if (conn && conn.open) {
    try {
      const envelope = await encryptPayload(conv.sharedKeyB64, {id: msgId, text, time: msg.time});
      conn.send({type: "direct_msg", enc: envelope});
    } catch(e) {
      addChatLine("System", "Failed to encrypt message. Check your connection.", true);
    }
  } else {
    msg.status = "failed";
    saveConvos();
    addChatLine("System", "Peer is offline. Message saved — resend when they reconnect.", true);
    tryConnect(activeConvId);
  }

  input.focus();
}
